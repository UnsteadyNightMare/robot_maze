import random
import numpy as np
actions = ['u','r','d','l']
qline = {'u':1.2, 'r':-2.1, 'd':-24.5, 'l':27}
epsilon = 0.3 # 0.3
def choose_action(epsilon):
    probs = np.ones(len(actions)) / len(actions)
    print(epsilon)
    random_num = random.random()
    if  random_num< epsilon: # 
        probs = np.ones(len(actions)) / len(actions) # 
        print(random_num)
        action = np.random.choice(actions, p=probs)
    else: 
        probs = np.zeros(len(actions)) #  Q 
        key = sorted(qline, key=lambda x:qline[x])[-1]
        for i, v in enumerate(actions):
            if v == key:
                probs[i] = 1 
        print(probs)
        print(random_num)
        action = np.random.choice(actions, p=probs)
    return action
for i in range(10):
    epsilon = float(0.3 / (i+1))
    print(choose_action(epsilon))
